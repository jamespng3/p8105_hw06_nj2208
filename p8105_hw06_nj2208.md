``` r
bw_mod = birthweight %>% 
  mutate(
    babysex = case_when(
      babysex == 1 ~"male",
      babysex == 2 ~"female"),
    frace = case_when(
      frace == 1 ~"white",
      frace == 2 ~"black",
      frace == 3 ~"asian",
      frace == 4 ~"puerto_rican",
      frace == 8 ~"other",
      frace == 9 ~"unknown"),
    mrace = case_when(
      mrace == 1 ~"white",
      mrace == 2 ~"black",
      mrace == 3 ~"asian",
      mrace == 4 ~"puerto_rican",
      mrace == 8 ~"other"),
    malform = case_when(
      malform == 0 ~"absent",
      malform == 1 ~"present")
    )
```

``` r
bw_guess1 = lm (bwt ~ ., data=bw_mod)%>% 
  broom::tidy()

bw_guess2 = lm (
  bwt ~ babysex + bhead + blength + delwt +gaweeks + parity + smoken, 
  data=bw_mod)
```

``` r
guess = bw_mod %>% 
  modelr::add_residuals(bw_guess2) %>%  
  modelr::add_predictions(bw_guess2)
guess %>% 
ggplot(aes(x=resid,y=pred))+
  geom_point()
```
